int n,res = -INF;
vector<vector<pii>> w(5);
bool vist[20202];

void dfs(int dep, int S){
	if(dep == 5){
		res = max<int>(res, S);
		return;
	}
	for(int i = 0; i < 5; i++){
		auto [a, b] = w[dep][i];
		if(vist[b]) continue;
		vist[b] = 1;
		dfs(dep + 1, S + a);
		vist[b] = 0;
	}
}

void solve(){
    cin >> n;
    for(int i = 1; i <= n; i++){
        for(int j = 0; j < 5; j++){
            int x; cin >> x;
			w[j].push_back({x, i});
		}
    }
	for(int j = 0; j < 5; j++) sort(all(w[j]), [&](pii& a, pii& b){
		return a.first > b.first;
	});
	dfs(0, 0);
	cout << res << "\n";
}

void Main(){
	int t = 1;
	//ri(t);
	while(t--) solve();
