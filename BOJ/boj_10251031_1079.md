int n,res, x, R[22][22];
vi v(22);
bool vist[22];

void f(int alive, int cnt, int flag = 0){
	if(flag or alive == 0){
		res = max<int>(res, cnt);
		return;
	}
	if(alive & 1){
		vector<pii> w;
		for(int j = 0; j < n; j++){
			if(vist[j]) continue;
			w.pb({v[j], j});
		}
		sort(all(w), [&](pii& a, pii& b){
			return a.first == b.first ? a.second < b.second : a.first > b.first;
		});
		auto [mx, mxidx] = w[0];
		vist[mxidx] = 1;
		f(alive - 1, cnt, x == mxidx);
		vist[mxidx] = 0;
	}
	else{
		for(int i = 0; i < n; i++){
			if(i == x or vist[i]) continue;
			vist[i] = 1;
			for(int j = 0; j < n; j++){
				if(vist[j]) continue;
				v[j] += R[i][j];
			}
			f(alive - 1, cnt + 1, flag);
			for(int j = 0; j < n; j++){
				if(vist[j]) continue;
				v[j] -= R[i][j];
			}
			vist[i] = 0;
		}
	}
}

void solve(){
	ri(n);
	for(int i = 0; i < n; i++) ri(v[i]);
	for(int i = 0; i < n; i++) for(int j = 0; j < n; j++) ri(R[i][j]);
	ri(x); 
	f(n, 0);
	po(res);
}
