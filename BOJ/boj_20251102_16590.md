# 16590번 : KMP

문제 제목이 KMP길래 KMP인가...생각했는데 이분매칭이였다.
각 이름의 대문자들만 기억해서 이분 그래프로 모델링을 하면 된다.
왼쪽 정점은 알파벳의 개수로, 오른쪽 정점들은 사람의 수$(N)$으로 놓고 입력으로 들어온 이름들의 앞글자만 기억해서 알파벳 -> 사람번호로 그래프를 모델링 한 다음에 $ Q$개의 쿼리를 Naive하게 처리해주면 된다.
답이 $NO$가 되는 경우는 1. 매칭의 결과가 입력으로 들어온 문자열의 수와 다르거나, 어떤 알파벳이 k번 들어왔을 때 k번 연결할 수 없으면 $NO$를 출력하면 된다.

int n,m;

void solve(){
	ri(n, m); cin.ignore();
	vvi g(27);
	for(int i = 0; i < n; i++){
		string s; getline(cin, s);
		vs v;
		stringstream in(s); string temp;
		bool cnt[26]{};
		while(in >> temp) v.pb(temp);
		for(const auto& c : v){
			string s = c;
			int b = s[0] - 'A';
			cnt[b] = 1;
		}
		for(int j = 0; j < 26; j++) if(cnt[j]) g[j].push_back(i);
	}
	vi par(n + 1, -1), vist(n);
	auto dfs = [&](int cur, auto&& dfs) -> bool{
		for(const auto& nxt : g[cur]){
			if(vist[nxt]) continue;
			vist[nxt] = 1;
			if(par[nxt] == -1 or dfs(par[nxt], dfs)){
				par[nxt] = cur;
				return 1;
			}
		}
		return 0;
	};
	while(m--){
		string s; ri(s);
		if(si(s) > n){
			po("NO");
			continue;
		}
		int cnt[26]{};
		for(auto c : s) cnt[c - 'A']++;
		bool ok = 1;
		for(int i = 0; i < 26; i++){
			if(cnt[i] > si(g[i])){
				ok = 0;
				break;
			}
		}
		if(!ok){
			po("NO");
			continue;
		}
		fill(all(par), -1);
		int res = 0;
		for(int i = 0; i < 26; i++){
			for(int j = 0; j < cnt[i]; j++){
				fill(all(vist), 0);
				if(dfs(i, dfs)) res++;
				else{
					ok = 0;
					break;
				}
			}
			if(!ok) break;
		}
		po((ok and res == si(s)) ? "YES" : "NO");
	}
}
